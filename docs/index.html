Schema Definitions

Users: object

username: string (1 to 150 chars)

Required. 150 characters or fewer. Letters, digits and @/./+/-/_ only.

password: string (1 to 128 chars)
email: string (email) (1 to 100 chars)
name: string (1 to 60 chars)
Example
{
  "username": "string",
  "password": "string",
  "email": "string (email)",
  "name": "string"
}

Requests: object

id: integer

Unique ID of request

lat: string (1 to 20 chars)

Latitude

lng: string (1 to 20 chars)

Longitude

location: string (1 to 60 chars)

Location name

name: string (1 to 50 chars)

Name of requestee

phone: string (1 to 20 chars)

Phone of requestee

request_for_others: boolean

Whether this request is not made for yourself

need_food: boolean
need_medicine: boolean
need_rescue: boolean
status: string (1 to 15 chars)

Status of request

Example
{
  "id": "integer",
  "lat": "string",
  "lng": "string",
  "location": "string",
  "name": "string",
  "phone": "string",
  "request_for_others": "boolean",
  "need_food": "boolean",
  "need_medicine": "boolean",
  "need_rescue": "boolean",
  "status": "string"
}
